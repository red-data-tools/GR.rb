def geterror
  line = GRCommons::Fiddley::MemoryPointer.new(:int)
  file = GRCommons::Fiddley::MemoryPointer.new(:pointer)
  e = super(1, line, file)
  return [0, nil, nil] if e == 0

  line = line.read_int
  # to_ptr: Fiddley::MemoryPointer -> Fiddle::Pointer
  file = file.to_ptr.ptr.to_s
  [e, line, file]
end
